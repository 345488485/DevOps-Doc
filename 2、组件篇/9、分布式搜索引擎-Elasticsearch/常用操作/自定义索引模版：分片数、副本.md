###### 自定义索引模版：分片数、副本

````shell
put _template/consumer-logs
{
    "index_patterns" : [
      "consumer-log-*"
    ],
    "settings" : {
      "number_of_replicas": 0,
      "number_of_shards": 4, 
      "index" : {
        "indexing" : {
          "slowlog" : {
            "threshold" : {
              "index" : {
                "warn" : "200ms",
                "trace" : "20ms",
                "debug" : "50ms",
                "info" : "100ms"
              }
            },
            "source" : "1000"
          }
        },
        "search" : {
          "slowlog" : {
            "threshold" : {
              "fetch" : {
                "warn" : "200ms",
                "trace" : "50ms",
                "debug" : "80ms",
                "info" : "100ms"
              },
              "query" : {
                "warn" : "500ms",
                "trace" : "50ms",
                "debug" : "100ms",
                "info" : "200ms"
              }
            }
          }
        },
        "unassigned" : {
          "node_left" : {
            "delayed_timeout" : "5m"
          }
        }
      }
    },
    "mappings" : { },
    "aliases" : { }
}
````
